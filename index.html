JS Promise


function getPromiseJson(url) {
    return new Promise(function(resolve, reject) {
        getJson(url, resolve);
        //getJson(url, function(response) {
        //    resolve(response);
        //})
    });
}


getPromiseJson("a.json")
    .then(function(a) {
        return getPromiseJson(a.url)
    })
    .then(function(b) {
        console.time(b);
    })
    .catch(function(error) {
        console.time("Error!", error);
    })

Get A and B in parallel


var arrayOfPromises = [
    getPromiseJson("a.json"),
    getPromiseJson("b.json")
];

Promise.all(arrayOfPromises).then(function(arrayOfResults) {
    console.time(arrayOfResults);
});

Deferred

  • Another way to create promise
  • Doesn't have native support

function getPromiseJson(url) {
    return new Promise(function(resolve, reject) {
        getJson(url, resolve)
    });
}
                

//https://github.com/tildeio/rsvp.js
function getPromiseJson(url) {
    var deferred = RSVP.defer();

    getJson(url, deferred.resolve);

    return deferred.promise;
}
